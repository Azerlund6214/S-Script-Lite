	// Дописать докачку (считаем колво while)
	function D__DOWNLOAD_WITH_PARTS()
	{
		if( !$fp = fopen("http://www.zappos.com/nike-free-run-black-victory-green-anthracite-white?zlfid=111" ,"r" ))
			return false;
			
		$content = "";

		while(!feof($fp)) 
		{ //
			$content .= fgets($fp, 1024);
		}
		fclose($fp); //закрываем
		

		$this -> SHD_Current_HTML = str_get_html( $content ); //SHD-функция
		
	}
